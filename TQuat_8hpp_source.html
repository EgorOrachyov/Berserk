79  explicit TQuat(const Mat &mat) {
80  T q[4];// notation: x[0] y[1] z[2] w[3]
81  auto &v = mat.values;
82 
83  T trace = v[0] + v[5] + v[10];
84 
85  // Matrix 4x4 indexes
86  // 0 1 2 3
87  // 4 5 6 7
88  // 8 9 10 11
89  // 12 13 14 15
90 
91  // Check the diagonal
92  if (trace > 0.0) {
93  // positive diagonal
94 
95  T s = MathUtils::Sqrt(trace + 1.0f);
96  q[3] = static_cast<T>(s * 0.5);
97 
98  T t = 0.5f / s;
99  q[0] = (v[9] - v[6]) * t;
100  q[1] = (v[2] - v[8]) * t;
101  q[2] = (v[4] - v[1]) * t;
102  } else {
103  // negative diagonal
104 
105  int32 i = 0;
106  if (v[5] > v[0]) i = 1;
107  if (v[10] > v[4 * i + i]) i = 2;
108 
109  static const int32 NEXT[3] = {1, 2, 0};
110  int32 j = NEXT[i];
111  int32 k = NEXT[j];
112 
113  auto s = (T) MathUtils::Sqrt(v[i * 4 + i] - (v[j * 4 + j] + v[k * 4 + k]) + 1.0);
114 
115  T t;
116  if (s == 0.0) t = s;
117  else
118  t = static_cast<T>(0.5 / s);
119 
120  q[i] = s * 0.5f;
121  q[3] = (v[k * 4 + j] - v[j * 4 + k]) * t;
122  q[j] = (v[j * 4 + i] - v[i * 4 + j]) * t;
123  q[k] = (v[k * 4 + i] - v[i * 4 + k]) * t;
124  }
125 
126  scalar = q[3];
127  vec = {q[0], q[1], q[2]};
128  }

238  Mat AsMatrix() const {
239  T x = vec[0];
240  T y = vec[1];
241  T z = vec[2];
242 
243  T xx = x * x;
244  T xy = x * y;
245  T xz = x * z;
246  T xw = x * scalar;
247 
248  T yy = y * y;

251 
252  T zz = z * z;
253  T zw = z * scalar;
254 
255  Mat r;
256  auto &v = r.values;
257 
258  v[0] = 1 - 2 * (yy + zz);
259  v[1] = 2 * (xy - zw);
260  v[2] = 2 * (xz + yw);

263  v[5] = 1 - 2 * (xx + zz);
264  v[6] = 2 * (yz - xw);
265  v[7] = 0;
266  v[8] = 2 * (xz - yw);
267  v[9] = 2 * (yz + xw);
268  v[10] = 1 - 2 * (xx + yy);
269  v[11] = 0;

323  static TQuat Slerp(T t, const TQuat &a, const TQuat &b) {
324  T ang = angle(a, b);
325 
326  if (ang <= MathUtils::THRESH_FLOAT32) {
327  return lerp(t, a, b);
328  }
329 
330  TQuat r;
331  T angleSin = MathUtils::Sin(ang);
332  T angle1 = MathUtils::Sin(ang * (1 - t)) / angleSin;
333  T angle2 = MathUtils::Sin(ang * t) / angleSin;
