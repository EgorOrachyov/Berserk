361 inline uint32 GetRHIShaderDataTypeSize(RHIShaderDataType type) {
362  const uint32 fBase = 4;
363  const uint32 iBase = 4;
364  const uint32 uBase = 4;
365  const uint32 bBase = 4;
366 
367  switch (type) {
368  case RHIShaderDataType::Float1:
369  return fBase * 1;
370  case RHIShaderDataType::Float2:
371  return fBase * 2;
372  case RHIShaderDataType::Float3:
373  return fBase * 3;
374  case RHIShaderDataType::Float4:
375  return fBase * 4;
376  case RHIShaderDataType::Int1:
377  return iBase * 1;
378  case RHIShaderDataType::Int2:
379  return iBase * 2;
380  case RHIShaderDataType::Int3:
381  return iBase * 3;
382  case RHIShaderDataType::Int4:
383  return iBase * 4;
384  case RHIShaderDataType::Uint1:
385  return uBase * 1;
386  case RHIShaderDataType::Uint2:
387  return uBase * 2;
388  case RHIShaderDataType::Uint3:
389  return uBase * 3;
390  case RHIShaderDataType::Uint4:
391  return uBase * 4;
392  case RHIShaderDataType::Bool1:
393  return bBase * 1;
394  case RHIShaderDataType::Bool2:
395  return bBase * 2;
396  case RHIShaderDataType::Bool3:
397  return bBase * 3;
398  case RHIShaderDataType::Bool4:
399  return bBase * 4;
400  case RHIShaderDataType::Mat2:
401  return fBase * 2 * 2;
402  case RHIShaderDataType::Mat3:
403  return fBase * 3 * 3;
404  case RHIShaderDataType::Mat4:
405  return fBase * 4 * 4;
406  default:
407  return 0;
